💻 كتابة Array في الكود

❌ الطريقة القديمة (7 أسطر)

int a = 2; int b = 3; int c = 4; int d = 5; int e = 6; int f = 7; int g = 8;

✅ الطريقة الجديدة (سطر واحد!)

int pins[] = {2, 3, 4, 5, 6, 7, 8};
✨ من 7 أسطر إلى سطر واحد فقط!

⚠️ قاعدة ذهبية في Array

🎯

الأرقام تبدأ من 0 مش من 1!

📊 ترتيب الـ segments:

  • pins[0] → segment A (Pin 2)
  • pins[1] → segment B (Pin 3)
  • pins[2] → segment C (Pin 4)
  • pins[3] → segment D (Pin 5)
  • pins[4] → segment E (Pin 6)
  • pins[5] → segment F (Pin 7)
  • pins[6] → segment G (Pin 8)

💡 عايز تشغل segment C؟ اكتب: pins[2]

🎯 استخدام Array في الكود

// تجميع كل الـ pins في Array int pins[] = {2, 3, 4, 5, 6, 7, 8}; void setup() { // تحديد كل الـ pins كـ OUTPUT for(int i = 0; i < 7; i++) { pinMode(pins[i], OUTPUT); } } void loop() { // تشغيل segment A digitalWrite(pins[0], HIGH); }

💡 لتشغيل B:

digitalWrite(pins[1], HIGH);

💡 لتشغيل G:

digitalWrite(pins[6], HIGH);

🎯 المطلوب:

int pins[] = {2, 3, 4, 5, 6, 7, 8}; void setup() { for(int i = 0; i < 7; i++) { pinMode(pins[i], OUTPUT); } } void loop() { // شغّل كل segment واحد واحد for(int i = 0; i < 7; i++) { digitalWrite(pins[i], HIGH); delay(500); } }

🎯 المشروع: عداد ذكي من 0 لـ 9

💡 الفكرة:

نعمل Array فيه كل الأرقام من 0 لـ 9
ونعرضهم بالتتابع باستخدام For Loop!

📊 Array الأرقام:

كل رقم = 7 قيم (شغال/مطفي)
1 = شغال ✅ | 0 = مطفي ❌

الرقم 0

1,1,1,1,1,1,0

الرقم 1

0,1,1,0,0,0,0

الرقم 8

1,1,1,1,1,1,1

💻 كود المشروع - الجزء الأول

// تجميع الـ pins في Array int pins[] = {2, 3, 4, 5, 6, 7, 8}; // Array ثنائي الأبعاد فيه كل الأرقام int numbers[10][7] = { {1,1,1,1,1,1,0}, // الرقم 0 {0,1,1,0,0,0,0}, // الرقم 1 {1,1,0,1,1,0,1}, // الرقم 2 {1,1,1,1,0,0,1}, // الرقم 3 {0,1,1,0,0,1,1}, // الرقم 4 {1,0,1,1,0,1,1}, // الرقم 5 {1,0,1,1,1,1,1}, // الرقم 6 {1,1,1,0,0,0,0}, // الرقم 7 {1,1,1,1,1,1,1}, // الرقم 8 {1,1,1,1,0,1,1} // الرقم 9 };

💻 كود المشروع - الجزء الثاني

void setup() { // تحديد كل الـ pins كـ OUTPUT for(int i = 0; i < 7; i++) { pinMode(pins[i], OUTPUT); } } void loop() { // عرض الأرقام من 0 لـ 9 for(int num = 0; num < 10; num++) { // عرض الرقم الحالي for(int i = 0; i < 7; i++) { digitalWrite(pins[i], numbers[num][i]); } delay(1000); // استنى ثانية } }
🎉 عداد ذكي من 0 لـ 9 بكود نظيف ومنظم!
